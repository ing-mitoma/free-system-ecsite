resources/ts/
├── main.tsx             # 頂点（さっき提示してくれたコード）
├── router.tsx           # 🎛️ URLの行き先リスト（URLとページを紐付ける）
│
├── layouts/             # 🖼️ 画面の「共通枠」を入れるフォルダ
│   ├── ShopLayout.tsx   # 一般画面用（ヘッダー・フッター付き）
│   └── AdminLayout.tsx  # 管理画面用（左側にサイドバー付き）
│
├── pages/               # 📄 各画面の「メインコンテンツ」を入れるフォルダ
│   ├── Shop/            # 🛒 一般ユーザー向け画面
│   │   ├── Home.tsx     # トップページ（商品が並ぶ）＜アウトプット用＞
│   │   ├── Product.tsx  # 商品詳細ページ（「カートに入れる」がある）
│   │   └── Cart.tsx     # カート画面（購入手続きへ進む）
│   └── Admin/           # 📊 管理者向け画面
│       ├── Dashboard.tsx# ダッシュボード（売上グラフなど）
│       └── Products.tsx # 商品管理（新規登録フォームなど）
│
└── components/          # 🧱 使い回す小さな部品（カードやボタンなど）
    ├── ProductCard.tsx  # 商品を綺麗に見せるカード（Chakra UI）
    └── Header.tsx       # 一般画面用のナビゲーションバー
